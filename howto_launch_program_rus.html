How-to (howto) — английское выражение (англ. how to — как сделать), получившее распространение во многих языках, которым называют набор инструкций (часто кратких и неформальных) для выполнения каких-либо задач. Значительное количество инструкций в стиле how-to можно найти в интернете. Некоторые из ресурсов, содержащих how-to, основаны на вики-движке. В Linux-сообществе принято написание «HOWTO», поскольку это помогает найти нужную информацию в поисковых системах типа Google, не отвлекаясь на вхождения слов «how» и «to». Как правило, самая подробная информации о какой-то стороне работы с Linux находится в HOWTO.

Как Запустить PHP Программу используя JavaScript

Привет.

It is Possible to Launch PHP Program by Passing Parameters from Javascript Program using GET and POST Methods, Both are Described Below.

Both Methods use removeBom function, this function remove unnecessary BOM (Byte Order Mask) from the Beginning of the PHP Responce, this is Some Bug, i have no Idea Where it is in the Responce From....

Difference Between GET and POST methods is following - GET Method is Reasonable to Use if Length of Parameters, send by GET is relatively small (<1000 Characters), and POST is is Reasonable to Use if Length of Parameters is big (>1000 Characters).

1. Get Method, The JavaScript File: 

function launchProgram(pattern) {
	if (window.XMLHttpRequest) {
		// code for IE7+, Firefox, Chrome, Opera, Safari
		xhr=new XMLHttpRequest();
	} else {  // code for IE6, IE5
		xhr=new ActiveXObject("Microsoft.XMLHTTP");
	}
	xhr.onreadystatechange = function(){
		if (this.readyState==4 && this.status==200) {

			dir = JSON.parse(removeBom(this.responseText));
			// further processing here
		}
	};
	xhr.open("GET","scripts/php/dir.php?q="+"../../"+encodeURIComponent(pattern),true);
	xhr.send();
}


2. Post Method, The JavaScript File:. 

function launchProgram(pattern) {
	if (window.XMLHttpRequest) {
		// code for IE7+, Firefox, Chrome, Opera, Safari
		xhr=new XMLHttpRequest();
	} else {  // code for IE6, IE5
		xhr=new ActiveXObject("Microsoft.XMLHTTP");
	}
	xhr.onreadystatechange = function(){
		if (this.readyState==4 && this.status==200) {

			dir = JSON.parse(removeBom(this.responseText));
			// further processing here
		}
	};

	xhr.open("POST", "scripts/php/dir.php", true);
	xhr.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
	xhr.send("q=../../"+ encodeURIComponent(pattern));
}


Also the removeBom() function, add this to The JavaScript File. 

function removeBom(str) {
	var ch, st, re = [], j=0;
	for (var c = 0; c < str.length; c++ ) {
		if (c==5) return str; // nothing found
		ch = str.charCodeAt(c);
		if(ch < 127)
		{
			re[j++] = ch & 0xFF;
			if (removeBomCheckSoFar(re)) return removeBom(str.substr(c+1));
		}
		else
		{
			st = [];    // clear stack
			do {
				st.push( ch & 0xFF );  // push byte to stack
				ch = ch >> 8;          // shift value down by 1 byte
			}
			while ( ch );
			// add stack contents to result
			// done because chars have "wrong" endianness
			st = st.reverse();
			for(var k=0;k<st.length; ++k)
				re[j++] = st[k];
			if (removeBomCheckSoFar(re)) return removeBom(str.substr(c+1));
		}
	}
	return str; // nothing found
}

function removeBomCheckSoFar(re) {
	// re - an array of bytes
	
	// UTF-8 - EF BB BF - 239 187 191
	// UTF-16 (BE) - FE FF - 254 255
	// UTF-16 (LE) - FF FE - 255 254
	// UTF-32 (BE) - 00 00 FE FF - 0 0 254 255
	// UTF-32 (LE) - FF FE 00 00 - 255 254 0 0
	// UTF-7 - 1 - 2B 2F 76 38 - 43 47 118 56
	// UTF-7 - 2 - 2B 2F 76 39 - 43 47 118 57
	// UTF-7 - 3 - 2B 2F 76 2B - 43 47 118 43
	// UTF-7 - 4 - 2B 2F 76 2F - 43 47 118 47
	// UTF-7 - 5 - 2B 2F 76 38 2D - 43 47 118 56 45
	// UTF-1 - F7 64 4C - 247 100 76
	// UTF-EBCDIC - DD 73 66 73 - 221 115 102 115
	// SCSU - 0E FE FF - 14 254 255
	// BOCU-1 - FB EE 28 - 251 238 40
	// GB-18030 - 84 31 95 33 - 132 49 149 51
	

	if (typeof re[1] !== 'undefined') { // first 2 bytes exists
		if (re[0]==254 && re[1]==255) return 1; // UTF-16 (BE)
		if (re[0]==255 && re[1]==254) return 1; // UTF-16 (LE)
	} 
	if (typeof re[2] !== 'undefined') { // first 3 bytes exists
		if (re[0]==239 && re[1]==187 && re[2]==191) return 1; // UTF-8
		if (re[0]==247 && re[1]==100 && re[2]==76) return 1; // UTF-1
		if (re[0]==14 && re[1]==254 && re[2]==255) return 1; // SCSU
		if (re[0]==251 && re[1]==238 && re[2]==40) return 1; // BOCU-1
	}
	if (typeof re[3] !== 'undefined') { // first 4 bytes exists
		if (re[0]==0 && re[1]==0 && re[2]==254 && re[3]==255) return 1; // UTF-32 (BE)
		if (re[0]==255 && re[1]==254 && re[2]==0 && re[3]==0) return 1; // UTF-32 (LE)
		if (re[0]==43 && re[1]==47 && re[2]==118 && re[3]==56) return 1; // UTF-7 - 1
		if (re[0]==43 && re[1]==47 && re[2]==118 && re[3]==57) return 1; // UTF-7 - 2
		if (re[0]==43 && re[1]==47 && re[2]==118 && re[3]==43) return 1; // UTF-7 - 3
		if (re[0]==43 && re[1]==47 && re[2]==118 && re[3]==47) return 1; // UTF-7 - 4
		if (re[0]==221 && re[1]==115 && re[2]==102 && re[3]==115) return 1; // UTF-EBCDIC
		if (re[0]==132 && re[1]==49 && re[2]==149 && re[3]==51) return 1; // GB-18030
	}
	if (typeof re[4] !== 'undefined') { // first 5 bytes exists
		if (re[0]==43 && re[1]==47 && re[2]==118 && re[3]==56 && re[4]==45) return 1; // UTF-7 - 5
	}
	return 0;
}

foreach (array_filter(glob($pattern), 'is_file')  as $filename) {
	// pathinfo:
	// ['basename'] - Name + ext
	// ['extension'] - ext
	// ['filename'] - Name
	// ['dirname'] - Directory with LEADING ../
	$out[] = pathinfo($filename);
}
